#include<bits/stdc++.h>
#define MAX_N 100

int max(int x,int y)
{
    if(x-y>0)
        return x;
    else return y;
}

int min(int x,int y)
{
    if(x-y>0)
        return y;
    else return x;
}

int main()
{
    int L,n;
    int minT,maxT;
    int a[MAX_N];  //ç”¨æ¥æ”¶é›†æ¯åªèš‚èšçš„ä¿¡æ¯

        scanf("%d",&a[i]);
    }
    for(int i=1;i<=n;i++)
    {
        minT = max(minT,min(a[i],L-a[i]));
        maxT = max(maxT,max(a[i],L-a[i]));
    }
    printf("min:%d max:%d",minT,maxT);
    return 0;
}
